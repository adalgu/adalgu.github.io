3.6.3 dplyr 핵심 동사

**#dplyr (1) 행을 선택하는 filter()**
#filter(df, 조건)  
filter(gapminder, country=='Korea, Rep.')
filter(gapminder, country=='Korea, Rep.' & year==2007)

gapminder %>% filter(country=='Korea, Rep.')  
gapminder %>% filter(country=='Korea, Rep.')  


**#dplyr (2) 행(관측치)을 정렬하는 arrange()**  
arrange(gapminder, year, country)  
gapminder %>% arrange(year, country)  

dplyr::arrange(gapminder, year, country)  
gapminder %>% dplyr::arrange(year, country)  

**#dplyr (3) 열(변수)을 선택하는 select()**  
select(gapminder, pop, gdpPercap)  
gapminder %>% select(pop, gdpPercap)  

**#dplyr (4) 변수를 변환하는 mutate()**  
gapminder %>%
  mutate(total_gdp = pop * gdpPercap,
        le_gdp_ratio = lifeExp / gdpPercap,
        lgrk = le_gdp_ratio * 100)

**#dplyr (5) 요약 통계량을 계산하는 summarize()**
gapminder %>%  
  summarize(n_obs = n(),  
            n_countries = n_distinct(country),  
            n_years = n_distinct(year),  
            med_gdpc = median(gdpPercap),  
            max_gdppc = max(gdpPercap))`

**#dplyr (6) 랜덤 샘플을 위한 sample_n()과 sample_frac()**  
sample_n(gapminder, 10) #정해진 숫자의 행을 랜덤 샘플링  
sample_frac(gapminder, 0.01) #1%의 련덤 샘플 출력  

**#dplyr (7) 고유한 행을 찾아내는 distinct()**

3.6.4 group_by()를 이용한 그룹 연산

명령어 세트를 그룹으로 나눈 후 연산 적용하는 것

각 대륙별 기대 수명의 중간값

library(gapminder)
library(dplyr)
gapminder %>%
  filter(year == 2007) %>%
  group_by(continent) %>%
  summarize(median(lifeExp))
## # A tibble: 5 x 2
##   continent `median(lifeExp)`
##   <fct>                 <dbl>
## 1 Africa                 52.9
## 2 Americas               72.9
## 3 Asia                   72.4
## 4 Europe                 78.6
## 5 Oceania                80.7

3.6.5 dplyr 명령의 공통점, 함수형 프로그래밍, 체이닝

library(gapminder)
d1 = filter(gapminder, year == 2007)
d2 = group_by(d1, continent)
d3 = summarize(d2, lifeExp = median(lifeExp))
arrange(d3, -lifeExp)
## # A tibble: 5 x 2
##   continent lifeExp
##   <fct>       <dbl>
## 1 Oceania      80.7
## 2 Europe       78.6
## 3 Americas     72.9
## 4 Asia         72.4
## 5 Africa       52.9
gapminder %>%
  filter(year==2007) %>%
  group_by(continent) %>%
  summarize(lifeExp = median(lifeExp)) %>%
  arrange(-lifeExp)
## # A tibble: 5 x 2
